```python
import re

def extract_key_phrases(text):
    # Tokenize text into sentences
    sentences = re.split(r'[.!?]', text)
    
    # Remove empty strings
    sentences = [sentence.strip() for sentence in sentences if sentence]
    
    # Extract nouns and verbs using NLTK library
    import nltk
    from nltk.tokenize import word_tokenize
    
    words =''.join(sentences).lower()
    tokenized_words = word_tokenize(words)
    tagged_words = nltk.pos_tag(tokenized_words)
    
    # Filter out non-nouns and non-verbs
    keywords = []
    for word, pos in tagged_words:
        if pos.startswith('NN') or pos == 'VB':
            keywords.append(word.lower())
            
    return set(keywords)

import re

def extract_keyphrases(text):
    # Tokenize text into sentences
    sentences = re.split(r'[.!?]',text)
    
    # Remove empty strings
    sentences = [sentence.strip() for sentence in sentences if sentence]
    
    # Extract nouns (keyphrases) using NLTK library
    import nltk
    from nltk.tokenize import word_tokenize
    
    nltk.download('punkt')
    keyphrases_AC_1=[]
    keyphrases_AC_2=[]

    for i,sentence in enumerate(sentences[:len(sentences)//2]):
        words=word_tokenize(sentence.lower())
        
        # Find noun chunks
        tagged_words=nltk.pos_tag(words)

        for word,pos in tagged_words:
            if pos.startswith('NN'):
                keyphrases_AC_1.append(word)
                
    for j,sentence in enumerate(sentences[len(sentences)//2+1:],start=len(sentences)//2 + 1 ):
        words=word_tokenize(sentence.lower())

        # Find noun chunks
        tagged_words=nltk.pos_tag(words)

        for word,pos in tagged_words:
            if pos.startswith('NN') :
                keyphrases_AC_2.append(word)
                
    return(keyphrases_AC_1,keyphrases_AC_2)

import re
from collections import defaultdict

def extract_keyphrases(text):
    # Tokenize text into sentences
    sentences = sent_tokenize(text)
    
    # Initialize dictionary to store keyphrase counts
    keyphrases_counts = defaultdict(int)

    # Loop through each sentence
    for s in sentences:
        words = word_tokenize(s.lower())
        
        # Remove stopwords (common words like 'the', 'and')
        stop_words = set(stopwords.words('english'))
        filtered_words = [word for word in words if word not in stop_words]
        
        # Extract bigrams (pairs of adjacent words) and trigrams (triples of adjacent words)
        ngrams = []
        for i in range(len(filtered_words)-1):            
            ngram = tuple([filtered_words[i], filtered_words[i+1]])
            ngrams.append(ngram)
                
        for gram in ngrams:
            keyphrases_counts[gram] += 1
    
    return dict(keyphrases_counts)
